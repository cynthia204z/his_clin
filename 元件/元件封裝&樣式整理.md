```html
<template>
  <!-- shadow="hover" -->
  <el-card :body-style="{ padding: '0px' }" class="card" shadow="hover">
    <div class="card-top">
      <div class="card-left" :class="leftClass">
        <div class="card-photo">
          <!-- <img :src="item.photo" /> -->
        </div>
        <p v-if="dataList.status" class="status">{{ dataList.status }}</p>
      </div>
      <div class="card-container">
        <div class="card-head">
          <div class="card-name">
            <span>{{ dataList.patName }}</span>
            <p class="card-item" style="margin-left:5px">
              {{ dataList.sexType }}
              ({{ yearsOld }}歲)
            </p>
          </div>
          <div>
            <slot name="head" />
          </div>
        </div>
        <div class="card-main">
          <span v-if="No" class="card-item card-item-left">
            序號&nbsp;&nbsp;
            <p>{{ No }}</p>
          </span>
          <span
                v-for="(item, index) in dataList.details"
                :key="index"
                class="card-item card-item-left"
                >
            {{ item.label }}&nbsp;&nbsp;
            <p>{{ item.value }}</p>
            <p v-if="item.disc" style="color:red">({{ item.disc }})</p>
          </span>
          <slot />
        </div>
      </div>
    </div>
    <div v-if="hasFooter" class="card-footer">
      <slot name="footer" />
    </div>
  </el-card>
</template>
<script>
  export default {
    name: 'PatCard',
    props: {
      data: {
        type: Object,
        default() {
          return {
            patName: '',
            birthDate: undefined,
            sexType: undefined,
            details: []
          }
        }
      },
      hasFooter: {
        type: Boolean,
        default: true
      },
      No: {
        type: Number,
        default: undefined
      }
    },
    data() {
      return {
        leftClass: this.addClass(),
        dataList: [],
        yearsOld: undefined
      }
    },
    created() {
      this.$nextTick(() => {
        this.getValue()
      })
    },
    methods: {
      getValue() {
        this.dataList = this.data
        this.yearsOld = this.$moment().diff(this.dataList.birthDate, 'years')
      },
      addClass() {
        switch (this.data.statusStyle) {
          case 'red':
            return 'status-red'
          case 'green':
            return 'status-green'
          case 'blue':
            return 'status-blue'
          case 'yellow':
            return 'status-yellow'
        }
      }
    }
  }
</script>
```

```html
<template>
  <div>
    <!-- 分頁式 -->
    <el-select
               v-if="page"
               ref="lov"
               v-model="valueInComponent"
               :placeholder="placeholder"
               :loading="loading"
               filterable
               clearable
               @keydown.enter.native="getDataByKeywords"
               @clear="selectClear"
               @visible-change="emitValue"
               @focus="selectClear"
               >
      <el-option
                 v-for="item in dataList"
                 :key="item.valueInComponent"
                 class="el-options__lov"
                 :label="item[optionLabel]"
                 :value="item[optionValue]"
                 >
        <div class="lov-options">
          <span v-for="(columeItem, index) in lovColumeList" :key="index" :index="index">
            {{ columeItem.label }}
            <p>{{ item[columeItem.content] }}</p>
          </span>
        </div>
      </el-option>
      <div>
        <pagination
                    :total="total1"
                    :page.sync="querylist.currentPage"
                    :limit.sync="querylist.pageSize"
                    @pagination="updateValue"
                    />
      </div>
    </el-select>

    <!-- 滾動加載式 -->
    <el-select
               v-if="scroll"
               ref="lov"
               v-model="valueInComponent"
               v-el-select-scroll="handleScroll"
               :placeholder="placeholder"
               :loading="false"
               filterable
               clearable
               @keydown.enter.native="getDataByKeywords"
               @clear="selectClear"
               @visible-change="emitValue"
               @focus="selectClear"
               >
      <!-- @change="changeBlur" -->
      <el-option
                 v-for="item in dataList"
                 :key="item.valueInComponent"
                 class="el-options__lov"
                 :label="item[optionLabel]"
                 :value="item[optionValue]"
                 >
        <div class="lov-options">
          <span v-for="(columeItem, index) in lovColumeList" :key="index" :index="index">
            {{ columeItem.label }}
            <p>{{ item[columeItem.content] }}</p>
          </span>
        </div>
      </el-option>
      <div
           style="width:100%;height:30px;text-align:center;line-height:30px;font-size:10pt;color:gray"
           >{{ massage }}</div>
    </el-select>
  </div>
</template>
<script>
  import request from '@/utils/request'

  export default {
    name: 'LovSelect',
    props: {
      placeholder: {
        type: String,
        default: '輸入關鍵字查詢'
      },
      type: {
        type: String,
        default: ''
      },
      // 選項標題、欄位名稱
      lovColumeList: {
        type: Array,
        default: () => []
      },
      // api url
      apiUrl: {
        type: String,
        default: ''
      },
      // 去後端當查詢參數的欄位(輸入框打哪個欄位的內容)
      lovQueryColume: {
        type: Object,
        default() {
          return {}
        }
      },
      // 選項顯示的文字根據哪個欄位
      optionLabel: {
        type: String,
        default: ''
      },
      // 選項實際上的值根據哪個欄位
      optionValue: {
        type: String,
        default: ''
      },
      parentValue: {
        type: String,
        default: ''
      }
    },
    data() {
      return {
        massage: '滾動加載更多',
        page: false,
        scroll: false,
        loading: false,
        query: {},
        dataList: [],
        querylist: {
          currentPage: 1,
          pageSize: 20,
          sort: 'asc',
          sidx: ''
        },
        total1: 0,
        queryColume: this.lovQueryColume
      }
    },
    computed: {
      valueInComponent: {
        get() {
          return this.parentValue
        },
        set(val) {
          this.$emit('update:parentValue', val)
        }
      }
    },
    mounted: function() {
      this.$nextTick(this.typeJudgment())
    },
    methods: {
      typeJudgment() {
        if (this.type === 'page') {
          this.page = true
          this.scroll = false
        }
        if (this.type === 'scroll') {
          this.page = false
          this.scroll = true
        }
      },
      // 更新值
      updateValue(pageinfo) {
        this.loading = true
        this.setQuery()
        request({
          url: this.apiUrl,
          method: 'get',
          data: this.query
        }).then((res) => {
          this.dataList = res.data.list
          this.total1 = res.data.pagination.total
          this.loading = false
        })
      },

      setQuery(searchValue) {
        if (searchValue) {
          const columeName = Object.keys(this.queryColume)[0]
          this.queryColume[columeName] = searchValue
        }
        const _query = {
          ...this.querylist,
          ...this.queryColume
        }

        // eslint-disable-next-line no-unused-vars
        for (const key in this.query) {
          delete this.query[key]
        }
        // eslint-disable-next-line no-unused-vars
        for (const key in _query) {
          if (Array.isArray(_query[key])) {
            this.query[key] = _query[key].join()
          } else {
            this.query[key] = _query[key]
          }
        }
      },

      getDataByKeywords(val) {
        this.$forceUpdate()
        this.querylist = {
          currentPage: 1,
          pageSize: 20,
          sort: 'asc',
          sidx: ''
        }
        let searchValue = val
        if (val instanceof KeyboardEvent) {
          searchValue = val.target.value
        }
        this.setQuery(searchValue)

        this.loading = true
        request({
          url: this.apiUrl,
          method: 'get',
          data: this.query
        }).then((res) => {
          this.dataList = res.data.list
          this.total1 = res.data.pagination.total
          this.loading = false
        })
      },
      selectClear() {
        this.dataList = []
        // this.valueInComponent = ''
      },
      emitValue() {
        const columeName = Object.keys(this.queryColume)
        for (let i = 0; i < this.dataList.length; i++) {
          if (this.valueInComponent === this.dataList[i][columeName]) {
            this.$emit('lovEvent', this.dataList[i])
            this.$emit('selectChange')
          }
        }
      },
      handleScroll(param, scroll) {
        // 跳頁
        // if (
        //   param &&
        //   this.querylist.currentPage < this.total1 / this.querylist.pageSize &&
        //   scroll < 1
        // ) {
        //   this.querylist.currentPage++;
        //   this.updateValue();
        // }
        // if (!param && this.querylist.currentPage > 1 && scroll < 1 && this.dataList.length > 7) {
        //   this.querylist.currentPage--;
        //   this.updateValue();
        // }
        // if (this.dataList.length <= 7 && this.querylist.currentPage > 1) {
        //   if (!param) {
        //     this.querylist.currentPage--;
        //     this.updateValue();
        //   }
        // }
        // 累積資料，沒有跳頁
        this.massage = '滾動加載更多'
        if (param && this.dataList.length < this.total1 && scroll < 1) {
          this.querylist.pageSize = this.querylist.pageSize + 20
          this.updateValue()
        }
        if (this.dataList.length === this.total1) {
          this.massage = '沒有更多資料'
        } else if (
          this.querylist.pageSize === 20 &&
          this.total1 < 20
        ) {
          this.massage = '沒有更多資料'
        }
      },
      changeBlur() {
        // 試圖解決 el-select 選中選項後 blur 方法無效(造成已選取的值無法刪除或修改，只能點 clear 按鈕)，嘗試未果
        // 此 bug 為 el-select filterable 比純 el-select 多一層 div 導致事件失效
        // this.$emit('blur')
        // this.$refs.lov.$el.querySelector('input').blur()
        // this.$refs.lov.$el.input.handleClearClick()
        // console.log(this.$refs.lov.$el.querySelector('span span i'));
        // console.log(this.$refs.lov.$el.querySelector('div input'));
        // this.$refs.lov.$el.querySelector('div input').handleClearClick()
      }
    }
  }
</script>
```
